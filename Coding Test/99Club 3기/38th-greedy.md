  public int solution(int n, int k, int[] enemy) {
      int answer = 0;
      int sum = 0;
      PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);

      for (int e : enemy) {
          answer++;
          sum += e;
          pq.add(e);

          if (sum <= n) continue;

          if (k > 0) {
              sum -= pq.remove();
              k--;
              continue;
          }

          answer--;
          break;
      }

      return answer;
  }
