  private int index = 0;
  private void hanoi(int n, int from, int to, int temp, int[][] answer) {
      if (n == 1) {
          answer[index++] = new int[]{from, to};
          return;
      }

      hanoi(n - 1, from, temp, to, answer);
      answer[index++] = new int[]{from, to};
      hanoi(n - 1, temp, to, from, answer);
  }
